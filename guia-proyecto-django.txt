#Preparar carpeta de trabajo

#Clonar Repositorio
    git clone url

#Crear Entorno
    python -m venv entorno

#Activar entorno
    posix
        source directorio/nombre_entorno/bin/activate

    cmd (WINDOWS)
        directorio\nombre_entorno\Scripts\activate.bat

    PowerShell (WINDOWS)
         directorio\nombre_entorno\\Scripts\Activate.ps1

#Instalar Django en el entorno
    pip install django

#Crear Archivo requirements.txt
    pip freeze > requirements.txt

#Crear proyecto django

    #si queremos en la carpeta actual
        django-admin startproject nombre_proyecto .
    
    #si queremos crear una carpeta y se cree el proyecto dentro.
        django-admin startproject nombre_proyecto

    # si queremos especificar el directorio
        django-admin startproject nombre_proyecto directorio

#Configuraciones iniciales

#Subir cambios al repositorio
    git add .
    git commit -m "crea archivos iniciales del proyecto"
    git push
